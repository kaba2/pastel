A complex number allows for the following syntax:

x, y and z are of type Type.

// Assignment operators

x = y;

// Arithmetic operators

x *= y;
x /= y;
x += y;
x -= y;

x * y;
x / y;
x + y;
x - y;

-x;

// Equality operators

x == y;
x != y;

// In-place functions

negate(x);
invert(x);

// Some standard functions

x = exp(z);
x = log(z);

// Geometric functions

x = sin(z);
x = cos(z);
x = tan(z);

x = asin(z);
x = acos(z);
x = atan2(z1, z2);
